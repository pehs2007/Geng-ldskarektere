                        var canvas = document.getElementById('BC');
                var context = canvas.getContext('2d');
                
                for(var x=0;x<200;x++)
                {
                        for(var y=0;y<200;y++)
                        {       
                                var i=0;
                                var cx=-0.68+x/1200;
                                var cy=-0.74+y/1200;
                                var zx=0;
                                var zy=0;     
                                
                                do
                                {
                                        var xt=zx*zy;
                                        zx=zx*zx-zy*zy+cx;
                                        zy=2*xt+cy;
                                        i++;
                                }
                                while(i<255&&(zx*zx+zy*zy)<4);
                                        
                                var color=i.toString(16);
                                context.beginPath();
                                context.rect(x*4, y*4, 4, 4);
                                context.fillStyle ='#'+color+color+color;
                                context.fill();
                        }
                }
